\testfile{pgfplotstest.reverseaxis.tex}
{
	\pgfplotsset{compat=newest}

	\testsection{x dir=reverse}
	\begin{tikzpicture}
		\begin{axis}[
			x dir=reverse,
			title=A title,
			xlabel=xlabel,
			ylabel=ylabel,
			colorbar horizontal,
			colorbar style={x dir=reverse},
		]
		\addplot+[mesh,scatter,scatter src=x] {x^10};
		\end{axis}
	\end{tikzpicture}
