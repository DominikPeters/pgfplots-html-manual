\documentclass[a4paper]{article}

\usepackage{pgfplots}

\pgfplotsset{
	compat=1.8,
}

\begin{document}

	%\tracingcommands=2\tracingmacros=2
	
\def\image[#1]{%
	\begin{tikzpicture}
		\begin{axis}[
			xmin=-2,xmax=8,
			ymin=-3,
			ymax=2,
			xlabel=xlabel,
			ylabel=ylabel,
			zlabel=zlabel,
			%show outer normals,
			#1,
			tiny,
		]
		\end{axis}
	\end{tikzpicture}
}%


\foreach \X in {box,left,center,right,none} {
	\foreach \Y in {box,left,center,right,none} {
		\image[
			title={x=\X, y=\Y},
			axis x line=\X,
			axis y line=\Y,
		]
	}
	\par
}


\end{document}
