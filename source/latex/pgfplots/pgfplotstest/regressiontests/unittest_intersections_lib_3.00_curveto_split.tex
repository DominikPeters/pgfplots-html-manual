\def\showPathSegment#1#2{%
	\begin{tikzpicture}
		\draw[help lines] (0,0) grid[step=20pt] (200pt,100pt);

		\pgfsetcolor{blue}
		\pgfsetpathandBB{#2}%
		\pgfusepath{stroke}

		\pgfsetlinewidth{1pt}
		\pgfsetcolor{red}
		\pgfsetpathandBB{#1}%
		\pgfusepath{stroke}
	\end{tikzpicture}
}

\def\stripmoveto\pgfsyssoftpath@movetotoken#1\relax{%

}
