}%

\def\CS#1 cs={%
	\pgfutil@ifundefined{tikz@parse@cs@#1}{%
		\pgfplots@error{no such cs:#1}%
		\def\next{}
	}{%
		\def\next{\csname tikz@parse@cs@#1\endcsname}%
	}%
	\next
}
