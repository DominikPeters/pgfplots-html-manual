\pgfkeys{/pgfplots/height=4cm,/pgfplots/width=6cm}
Regular axis composition:

\begin{tikzpicture}
  \begin{axis}
    \addplot coordinates {
        (0,1000)
        (1,1100)
        (2,1200)
        (3,1300)
    };
  \end{axis}
\end{tikzpicture}

With scaled

\begin{tikzpicture}
  \begin{axis}[change y base,y SI prefix=kilo,change x base,x SI prefix=centi]
    \addplot coordinates {
        (0.01,1000)
        (0.02,1100)
        (0.03,1200)
        (0.04,1300)
    };
    \addlegendentry{\pgfkeys{/pgfplots/x unit prefix} and \pgfkeys{/pgfplots/y unit prefix}}
  \end{axis}
\end{tikzpicture}

With prefix no scale

\begin{tikzpicture}
  \begin{axis}[y SI prefix=kilo,x SI prefix=centi,x unit=T,use units]
    \addplot coordinates {
        (0.01,1000)
        (0.02,1100)
        (0.03,1200)
        (0.04,1300)
    };
  \end{axis}
\end{tikzpicture}
