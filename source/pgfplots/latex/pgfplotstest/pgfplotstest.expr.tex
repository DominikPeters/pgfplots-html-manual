\testsubsection{sin(x)}
\begin{tikzpicture}
\begin{axis}
\addplot (\x,{sin(\x r)});
\end{axis}
\end{tikzpicture}

\testsubsection{$x^2$}
\begin{tikzpicture}
\begin{axis}
\addplot plot[domain=0:10] (\x,\x^2);
\end{axis}
\end{tikzpicture}

\def\testexpression#1#2{%
	\testsubsection{$#1$}
	\begin{tikzpicture}
		\begin{axis}[title={$#1$},#2]
			\addplot {#1};
		\end{axis}
	\end{tikzpicture}%
}%
\def\testgnuplot#1#2{%
	\begin{tikzpicture}
		\begin{axis}[title={$#1$},#2]
			\addplot gnuplot {#1};
		\end{axis}
	\end{tikzpicture}%
}%
