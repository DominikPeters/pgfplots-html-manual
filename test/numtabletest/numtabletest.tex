\testsection{Reading a tab-separated table with many entries}
\testsubsection{Reading it...}
\tracingmacros=2\tracingcommands=2
\numtableread{numtabletestdata} to \table
\tracingmacros=0\tracingcommands=0

\testsubsection{Quering colnames...}
\tracingmacros=2\tracingcommands=2
\numtablegetcolumnlist\table\to\cols
\listforeach\cols\as\col{Column ``\col''\par}%
\tracingmacros=0\tracingcommands=0

\testsubsection{Quering column content by column name}
\tracingmacros=2\tracingcommands=2
\numtablegetcolumnbyname L2\of\table\to\colcontent
\listforeach\colcontent\as\colc{Row data: ``\colc''\par}%
\tracingmacros=0\tracingcommands=0

\testsubsection{Quering column content by column name for all cols}
\numtablegetcolumnlist\table\to\cols
\listforeach\cols\as\col{\textbf{Column ``\col''}\par
	\numtablegetcolumnbyname\col\of\table\to\colcontent
	\listforeach\colcontent\as\colc{Row data: ``\colc''\par}%
}%
